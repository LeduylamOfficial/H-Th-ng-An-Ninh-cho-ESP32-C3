Chào mừng bạn đến với hướng dẫn xây dựng hệ thống TheLeduylamCloudVN. Phần này cung cấp một cái nhìn tổng quan trực quan về những gì hệ thống của bạn sẽ làm được. Dưới đây là mô phỏng bảng điều khiển trung tâm hiển thị dữ liệu thời gian thực từ các cảm biến môi trường (Nhiệt độ, Độ ẩm) và trạng thái an ninh (Mưa, Rung) được xử lý bởi bo mạch ESP32-C3.

Cơ chế hoạt động thông minh

  • 🚨 Cảnh báo Đa tầng: Còi sẽ tự động hú nếu phát hiện Rung (trộm), quá nhiệt (>75°C), có mưa hoặc khi bạn bấm nút thủ công trên App Blynk.
  • 🔇 Chế độ Im lặng (Mute): Nút điều khiển V12 trên app cho phép bạn tắt tiếng còi vật lý nhưng vẫn nhận được thông báo đẩy (Push Notification) trên điện thoại.
  • 📺 LCD Xoay vòng: Màn hình LCD 16x2 sẽ tự động chuyển đổi thông số hiển thị sau mỗi 2 giây, giúp bạn xem được toàn bộ 10 thông số mà không cần thao tác bấm nút.
  • 🛡️ Chống Offline: Hệ thống sử dụng lệnh Blynk.config đảm bảo chip không bị treo hoặc dừng hoạt động đọc cảm biến khi mạng WiFi tại nhà bị mất kết nối.

Cài đặt Môi trường & Mã Nguồn

Phần cuối cùng cung cấp các thiết lập bắt buộc trong phần mềm Arduino IDE để nạp code thành công cho mạch ESP32-C3, kèm theo đoạn mã nguồn C++ hoàn thiện của dự án.

1. Cấu hình Tools (Bắt buộc)

  • Board: ESP32C3 Dev Module (hoặc AirM2M CORE ESP32C3)
  • USB CDC On Boot: Enabled (Để xem Serial Monitor)
  • Flash Mode: DIO
  • Flash Size: 4MB
Mẹo nạp Code: Nếu báo lỗi kết nối, nhấn giữ nút BOOT trên mạch, cắm cáp USB, chờ máy tính nhận cổng COM rồi bấm nạp.

2. Thư viện yêu cầu (Libraries)

Vào Tools -> Manage Libraries và cài đặt các thư viện sau:

main.ino 
/* Điền thông tin Blynk của bạn vào đây */
#define BLYNK_TEMPLATE_ID   "Dien_ID_Cua_Ban"
#define BLYNK_TEMPLATE_NAME "Dien_Ten_Template"
#define BLYNK_AUTH_TOKEN    "Dien_Auth_Token_Cua_Ban"

#include <WiFi.h>
#include <BlynkSimpleEsp32.h>
#include <DHT.h>
#include <Wire.h>
#include <LiquidCrystal_I2C.h>

/* Điền thông tin WiFi của bạn vào đây */
char ssid[] = "Ten_WiFi_Cua_Ban";
char pass[] = "Mat_Khau_WiFi";

// Dinh nghia chan Pin ESP32-C3
#define LED_PIN     10
#define RELAY_PIN   7
#define DHTPIN      2
#define SOIL_PIN    3
#define VIB_PIN     5
#define NTC_PIN     6
#define RAIN_PIN    1
#define BUZZ_PIN    20
#define DHTTYPE     DHT11

DHT dht(DHTPIN, DHTTYPE);
LiquidCrystal_I2C lcd(0x27, 16, 2);
BlynkTimer timer;

bool relayState = false;
bool manualBuzz = false;
bool buzzEnable = true; 
int lcdPage = 0;
unsigned long lastNotify = 0;

// Dieu khien tu Blynk
BLYNK_WRITE(V12) { buzzEnable = param.asInt(); }
BLYNK_WRITE(V11) { manualBuzz = param.asInt(); }
BLYNK_WRITE(V2)  { digitalWrite(LED_PIN, param.asInt()); }
BLYNK_WRITE(V6)  { relayState = param.asInt(); digitalWrite(RELAY_PIN, relayState); }

void sendNotify(String msg) {
  if (millis() - lastNotify > 60000) { 
    Blynk.logEvent("alarm", msg); 
    lastNotify = millis();
  }
}

void checkAlarms() {
  float t = dht.readTemperature();
  bool isRain = !digitalRead(RAIN_PIN);
  bool isVib = !digitalRead(VIB_PIN);
  bool isHotNTC = !digitalRead(NTC_PIN); 

  bool trigger = false;
  String reason = "";

  if (isHotNTC || t >= 75) { trigger = true; reason = "NHIET DO QUA CAO!"; }
  if (isRain) { trigger = true; reason = "TROII DANG MUA!"; }
  if (isVib) { trigger = true; reason = "CANH BAO RUNG!"; }

  if ((trigger || manualBuzz) && buzzEnable) {
    digitalWrite(BUZZ_PIN, HIGH);
    if (trigger) sendNotify(reason);
  } else {
    digitalWrite(BUZZ_PIN, LOW);
  }

  Blynk.virtualWrite(V8, isVib ? "RUNG!" : "Safe");
  Blynk.virtualWrite(V9, isRain ? "MUA!" : "Dry");
  Blynk.virtualWrite(V10, isHotNTC ? 100 : 0);
}

void updateLCD() {
  float t = dht.readTemperature();
  float h = dht.readHumidity();
  int soil = map(analogRead(SOIL_PIN), 4095, 0, 0, 100);
  soil = constrain(soil, 0, 100);

  Blynk.virtualWrite(V3, t);
  Blynk.virtualWrite(V4, h);
  Blynk.virtualWrite(V5, soil);

  lcd.setCursor(0, 0);
  if (lcdPage == 0) {
    lcd.print("T:"); lcd.print((int)t); lcd.print("C H:"); lcd.print((int)h); lcd.print("%   ");
    lcd.setCursor(0, 1);
    lcd.print("Buzz:"); lcd.print(buzzEnable ? "ON " : "OFF"); 
    lcd.print(" R:"); lcd.print(relayState ? "ON " : "OFF");
    lcdPage = 1;
  } else {
    lcd.print("Soil:"); lcd.print(soil); lcd.print("% Rain:"); lcd.print(!digitalRead(RAIN_PIN)? "Y":"N");
    lcd.setCursor(0, 1);
    lcd.print("System Online   ");
    lcdPage = 0;
  }
}

void setup() {
  Serial.begin(115200);
  Wire.begin(8, 9);
  lcd.init(); lcd.backlight();
  
  pinMode(BUZZ_PIN, OUTPUT);
  pinMode(LED_PIN, OUTPUT);
  pinMode(RELAY_PIN, OUTPUT);
  pinMode(VIB_PIN, INPUT_PULLUP);
  pinMode(NTC_PIN, INPUT_PULLUP);
  pinMode(RAIN_PIN, INPUT_PULLUP);

  dht.begin();
  
  WiFi.begin(ssid, pass);
  Blynk.config(BLYNK_AUTH_TOKEN); 
  
  timer.setInterval(2000L, updateLCD);
  timer.setInterval(500L, checkAlarms);
}

void loop() {
  Blynk.run();
  timer.run();
}
